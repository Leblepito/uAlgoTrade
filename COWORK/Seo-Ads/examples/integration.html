Integration Code Snippets

JavaScript (fetch)

const resp = await fetch('https://your-seo-ads.railway.app/content/generate', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'X-API-Key': 'your-api-key'
  },
  body: JSON.stringify({
    region: 'tr',
    platform: 'meta_ig',
    topic: 'crypto_trading',
    symbol: 'BTCUSDT'
  })
});
const data = await resp.json();
console.log(data.pieces[0].headline);

Python (httpx)

import httpx

resp = httpx.post(
    "https://your-seo-ads.railway.app/content/generate",
    headers={"X-API-Key": "your-api-key"},
    json={"region": "tr", "platform": "meta_ig", "topic": "crypto_trading"}
)
data = resp.json()
print(data["pieces"][0]["headline"])

cURL

curl -X POST https://your-seo-ads.railway.app/content/generate \
  -H "Content-Type: application/json" \
  -H "X-API-Key: your-api-key" \
  -d '{"region":"tr","platform":"meta_ig","topic":"crypto_trading","symbol":"BTCUSDT"}'

React / Next.js

const [content, setContent] = useState(null);

async function generateContent(region, platform) {
  const res = await fetch('/api/seo-ads', {
    method: 'POST',
    body: JSON.stringify({ region, platform, topic: 'crypto_trading' })
  });
  setContent(await res.json());
}
